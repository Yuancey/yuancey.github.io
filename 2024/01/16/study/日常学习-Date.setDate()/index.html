Date用法

Date.setDate()

setDate() 方法用于设置一个月的某一天

基本适用:


var d = new Date();

console.log("---d---",d);   //2024-01-16T03:30:08.832Z

d.setDate(15);

console.log("---d---",d);   //2024-01-15T03:30:38.566Z

应用：可以设置某个日期的前几天，后几天

// 例：当前日期的前13天 |  近14天
function getDate(index) {
  const date = new Date();
  const newDate = new Date();
  newDate.setDate(date.getDate() + index);
  const beg = newDate.getFullYear() + "-" + ((newDate.getMonth() + 1 < 10 ? '0'+(newDate.getMonth() + 1): newDate.getMonth() + 1 )) + "-" + (newDate.getDate() < 10? '0' + newDate.getDate(): newDate.getDate());
  const end = date.getFullYear() + "-" + ((date.getMonth() + 1 < 10 ? '0'+(date.getMonth() + 1): date.getMonth() + 1 )) + "-" + (date.getDate() < 10? '0' + date.getDate(): date.getDate());
  return [beg, end];
}
console.log(getDate(-13));  //[ '2024-01-03', '2024-01-16' ]
