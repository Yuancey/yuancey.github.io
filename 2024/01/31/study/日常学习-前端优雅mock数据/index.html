function initMock() {
  ;[...topMockData, ...rightBottomMockData].forEach((item) => {
    let url = item.uri
    if (process.env.NODE_ENV === 'development') {
      url = api.defaults.baseURL + url
    }
    Mock.mock(url, item.method || 'post', item.data)
  })
}

export default initMock

import api from '@/api/api'
if (process.env.NODE_ENV === 'development') {
  // 模拟接口数据
  initMock()
}

// TODO: 上线前需要移除

  initMock()

function simplePost(uri, payload = {}) {
  return api.post(uri, payload).then((res) => {
    return res.data
  })
}

// 缺陷统计
export function getBugCount(projectId, time) {
  //待优化:一个接口报错，就会返回reject

      projectId: projectId,

    }),
    //待处理缺陷数（线上）
    simplePost('/api/portal/bugOnline', {
      projectId: projectId,
      month: time,
    }),
  ])
}
